<script>  
  
/* ================== 基本設定 ================== */  
  
let audioContext;  
let masterGain;  
let running=false;  
  
const noteNamesJP=["ド","ド#","レ","レ#","ミ","ファ","ファ#","ソ","ソ#","ラ","ラ#","シ"];  
const noteNamesEN=["C","C#","D","D#","E","F","F#","G","G#","A","A#","B"];  
  
let currentH=0,currentM=0,currentS=0;  
  
/* ================== 音処理 ================== */  
  
function initAudio(){  
    if(!audioContext){  
        audioContext=new (window.AudioContext||window.webkitAudioContext)();  
        masterGain=audioContext.createGain();  
        masterGain.connect(audioContext.destination);  
    }  
}  
  
function playTone(index,duration){  
    const osc=audioContext.createOscillator();  
    const gain=audioContext.createGain();  
    osc.frequency.value=261.63*Math.pow(2,index/12);  
    gain.gain.value=0.2;  
    osc.connect(gain);  
    gain.connect(masterGain);  
    osc.start();  
    osc.stop(audioContext.currentTime+duration);  
}  
  
/* ================== アナログ時計 ================== */  
  
const canvas=document.getElementById("analog");  
const ctx=canvas.getContext("2d");  
  
function drawClock(){  
    const now=new Date();  
    const w=canvas.width/2;  
    const h=canvas.height/2;  
    ctx.clearRect(0,0,300,300);  
  
    ctx.save();  
    ctx.translate(w,h);  
  
    ctx.beginPath();  
    ctx.arc(0,0,140,0,Math.PI*2);  
    ctx.strokeStyle="#444";  
    ctx.lineWidth=4;  
    ctx.stroke();  
  
    const hour=now.getHours()%12;  
    const minute=now.getMinutes();  
    const second=now.getSeconds();  
  
    drawHand((hour+minute/60)*30,60,6,"#ff3b3b");  
    drawHand(minute*6,90,4,"#3b8bff");  
    drawHand(second*6,110,2,"#00d27a");  
  
    ctx.restore();  
}  
  
function drawHand(angle,length,width,color){  
    ctx.save();  
    ctx.rotate(angle*Math.PI/180);  
    ctx.beginPath();  
    ctx.moveTo(0,0);  
    ctx.lineTo(0,-length);  
    ctx.strokeStyle=color;  
    ctx.lineWidth=width;  
    ctx.stroke();  
    ctx.restore();  
}  
  
/* ================== 音階ゾーン生成 ================== */  
  
function createTable(){  
    const table=document.getElementById("noteTable");  
    for(let i=11;i>=0;i--){  
        const row=document.createElement("tr");  
        row.innerHTML=`  
        <td>${i+1}</td>  
        <td id="H${i}">${noteNamesJP[i]}</td>  
        <td id="M${i}">${noteNamesJP[i]}</td>  
        <td id="S${i}">${noteNamesJP[i]}</td>`;  
        table.appendChild(row);  
    }  
}  
  
function gradientColor(base,index){  
    const intensity=(index+1)/12;  
    return `rgba(${base},${0.3+intensity*0.7})`;  
}  
  
function updateNoteZone(h,m,s){  
  
    currentH=h;  
    currentM=m;  
    currentS=s;  
  
    for(let i=0;i<12;i++){  
        ["H","M","S"].forEach(type=>{  
            const cell=document.getElementById(type+i);  
            cell.classList.remove("activeCell");  
            cell.style.background="";  
        });  
    }  
  
    for(let i=h+1;i<12;i++){  
        document.getElementById("H"+i).style.background=  
        `linear-gradient(to top, rgba(255,59,59,0.3), rgba(255,59,59,${0.3+(i/12)}))`;  
    }  
  
    for(let i=m+1;i<12;i++){  
        document.getElementById("M"+i).style.background=  
        `linear-gradient(to top, rgba(59,139,255,0.3), rgba(59,139,255,${0.3+(i/12)}))`;  
    }  
  
    for(let i=s+1;i<12;i++){  
        document.getElementById("S"+i).style.background=  
        `linear-gradient(to top, rgba(0,210,122,0.3), rgba(0,210,122,${0.3+(i/12)}))`;  
    }  
  
    document.getElementById("H"+h).classList.add("activeCell");  
    document.getElementById("M"+m).classList.add("activeCell");  
    document.getElementById("S"+s).classList.add("activeCell");  
}  
  
/* ================== メインループ ================== */  
  
function tick(){  
    if(!running)return;  
  
    const now=new Date();  
    document.getElementById("clock").innerText=now.toLocaleTimeString();  
  
    const h=now.getHours()%12;  
    const m=Math.floor(now.getMinutes()/5);  
    const s=Math.floor(now.getSeconds()/5);  
  
    updateNoteZone(h,m,s);  
  
    drawClock();  
  
    if(now.getMilliseconds()<30){  
        playTone(s,0.2);  
    }  
  
    requestAnimationFrame(tick);  
}  
  
function startApp(){  
    initAudio();  
    running=true;  
    tick();  
}  
  
function stopApp(){  
    running=false;  
}  
  
createTable();  
drawClock();  
  
</script>
